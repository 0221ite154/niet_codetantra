import sys
expr = sys.argv[1]
def evaluate(expr):
    # Write code here
    stack = []
    prev_op = None
    for token in expr.split():
        if token in ["+","-"]:
            op2 = stack.pop()
            op1 = stack.pop()
            stack.append(f"{op1} {token} {op2}")
        elif token in ["*", "/"]:
            op2 = stack.pop()
            op1 = stack.pop()
            if prev_op in ["+", "-"]:
                stack.append(f"({op1}) {token} ({op2})")
            else:
                stack.append(f"{op1} {token} {op2}")
        else:
            stack.append(token)
        prev_op = token
    return stack[0]
print(evaluate(expr))
