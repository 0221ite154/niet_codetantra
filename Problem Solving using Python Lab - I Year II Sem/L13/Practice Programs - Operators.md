```python
a=int(input("Enter first value: "))
b=int(input("Enter second value: "))

print("{} - {} =".format(a,b),a-b)
print("{} * {} =".format(a,b),a*b)
print("{} / {} =".format(a,b),a/b)
print("{} ** {} =".format(a,b),a**b)
print("{} % {} =".format(a,b),a%b)
print("{} // {} =".format(a,b),a//b)
```
